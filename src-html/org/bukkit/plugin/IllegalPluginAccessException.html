001    package org.bukkit.plugin;
002    
003    /**
004     * Thrown when a plugin attempts to interact with the server when it is not
005     * enabled
006     */
007    @SuppressWarnings("serial")
008    public class IllegalPluginAccessException extends RuntimeException {
009    
010        /**
011         * Creates a new instance of <code>IllegalPluginAccessException</code>
012         * without detail message.
013         */
014        public IllegalPluginAccessException() {}
015    
016        /**
017         * Constructs an instance of <code>IllegalPluginAccessException</code>
018         * with the specified detail message.
019         *
020         * @param msg the detail message.
021         */
022        public IllegalPluginAccessException(String msg) {
023            super(msg);
024        }
025    }
