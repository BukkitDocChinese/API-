001    package org.bukkit.event.painting;
002    
003    import org.bukkit.Warning;
004    import org.bukkit.entity.Painting;
005    import org.bukkit.event.Event;
006    
007    /**
008     * Represents a painting-related event.
009     *
010     * @deprecated Use {@link org.bukkit.event.hanging.HangingEvent} instead.
011     */
012    @Deprecated
013    @Warning(reason="This event has been replaced by HangingEvent")
014    public abstract class PaintingEvent extends Event {
015        protected Painting painting;
016    
017        protected PaintingEvent(final Painting painting) {
018            this.painting = painting;
019        }
020    
021        /**
022         * Gets the painting involved in this event.
023         *
024         * @return the painting
025         */
026        public Painting getPainting() {
027            return painting;
028        }
029    }
