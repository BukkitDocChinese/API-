001    package org.bukkit.event.player;
002    
003    import org.bukkit.entity.Player;
004    import org.bukkit.event.HandlerList;
005    import org.bukkit.event.inventory.InventoryClickEvent;
006    import org.bukkit.event.inventory.InventoryOpenEvent;
007    import org.bukkit.inventory.Inventory;
008    
009    /**
010     * Represents a player related inventory event; note that this event never
011     * actually did anything
012     *
013     * @deprecated Use {@link InventoryClickEvent} or {@link InventoryOpenEvent}
014     *     instead, or one of the other inventory events in {@link
015     *     org.bukkit.event.inventory}.
016     */
017    @Deprecated
018    public class PlayerInventoryEvent extends PlayerEvent {
019        private static final HandlerList handlers = new HandlerList();
020        protected Inventory inventory;
021    
022        public PlayerInventoryEvent(final Player player, final Inventory inventory) {
023            super(player);
024            this.inventory = inventory;
025        }
026    
027        /**
028         * Gets the Inventory involved in this event
029         *
030         * @return Inventory
031         */
032        public Inventory getInventory() {
033            return inventory;
034        }
035    
036        @Override
037        public HandlerList getHandlers() {
038            return handlers;
039        }
040    
041        public static HandlerList getHandlerList() {
042            return handlers;
043        }
044    }
