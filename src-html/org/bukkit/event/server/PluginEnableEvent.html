001    package org.bukkit.event.server;
002    
003    import org.bukkit.event.HandlerList;
004    import org.bukkit.plugin.Plugin;
005    
006    /**
007     * Called when a plugin is enabled.
008     */
009    public class PluginEnableEvent extends PluginEvent {
010        private static final HandlerList handlers = new HandlerList();
011    
012        public PluginEnableEvent(final Plugin plugin) {
013            super(plugin);
014        }
015    
016        @Override
017        public HandlerList getHandlers() {
018            return handlers;
019        }
020    
021        public static HandlerList getHandlerList() {
022            return handlers;
023        }
024    }
