001    package org.bukkit.event.server;
002    
003    import org.bukkit.command.CommandSender;
004    import org.bukkit.event.HandlerList;
005    
006    /**
007     * This event is called when a command is recieved over RCON. See the javadocs
008     * of {@link ServerCommandEvent} for more information.
009     */
010    public class RemoteServerCommandEvent extends ServerCommandEvent {
011        private static final HandlerList handlers = new HandlerList();
012    
013        public RemoteServerCommandEvent(final CommandSender sender, final String command) {
014            super(sender, command);
015        }
016    
017        @Override
018        public HandlerList getHandlers() {
019            return handlers;
020        }
021    
022        public static HandlerList getHandlerList() {
023            return handlers;
024        }
025    }
