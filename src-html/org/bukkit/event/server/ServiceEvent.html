001    package org.bukkit.event.server;
002    
003    import org.bukkit.plugin.RegisteredServiceProvider;
004    
005    /**
006     * An event relating to a registered service. This is called in a {@link
007     * org.bukkit.plugin.ServicesManager}
008     */
009    public abstract class ServiceEvent extends ServerEvent {
010        private final RegisteredServiceProvider<?> provider;
011    
012        public ServiceEvent(final RegisteredServiceProvider<?> provider) {
013            this.provider = provider;
014        }
015    
016        public RegisteredServiceProvider<?> getProvider() {
017            return provider;
018        }
019    }
