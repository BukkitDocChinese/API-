001    package org.bukkit.event.server;
002    
003    import org.bukkit.event.HandlerList;
004    import org.bukkit.plugin.RegisteredServiceProvider;
005    
006    /**
007     * This event is called when a service is registered.
008     * <p>
009     * Warning: The order in which register and unregister events are called
010     * should not be relied upon.
011     */
012    public class ServiceRegisterEvent extends ServiceEvent {
013        private static final HandlerList handlers = new HandlerList();
014    
015        public ServiceRegisterEvent(RegisteredServiceProvider<?> registeredProvider) {
016            super(registeredProvider);
017        }
018    
019        @Override
020        public HandlerList getHandlers() {
021            return handlers;
022        }
023    
024        public static HandlerList getHandlerList() {
025            return handlers;
026        }
027    }
