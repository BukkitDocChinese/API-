001    package org.bukkit.event.vehicle;
002    
003    import org.bukkit.entity.Vehicle;
004    import org.bukkit.event.HandlerList;
005    
006    /**
007     * Raised when a vehicle is created.
008     */
009    public class VehicleCreateEvent extends VehicleEvent {
010        private static final HandlerList handlers = new HandlerList();
011    
012        public VehicleCreateEvent(final Vehicle vehicle) {
013            super(vehicle);
014        }
015    
016        @Override
017        public HandlerList getHandlers() {
018            return handlers;
019        }
020    
021        public static HandlerList getHandlerList() {
022            return handlers;
023        }
024    }
