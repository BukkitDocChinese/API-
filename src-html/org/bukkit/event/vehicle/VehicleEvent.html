001    package org.bukkit.event.vehicle;
002    
003    import org.bukkit.entity.Vehicle;
004    import org.bukkit.event.Event;
005    
006    /**
007     * Represents a vehicle-related event.
008     */
009    public abstract class VehicleEvent extends Event {
010        protected Vehicle vehicle;
011    
012        public VehicleEvent(final Vehicle vehicle) {
013            this.vehicle = vehicle;
014        }
015    
016        /**
017         * Get the vehicle.
018         *
019         * @return the vehicle
020         */
021        public final Vehicle getVehicle() {
022            return vehicle;
023        }
024    }
