001    package org.bukkit.event.world;
002    
003    import org.bukkit.Chunk;
004    import org.bukkit.event.HandlerList;
005    import org.bukkit.generator.BlockPopulator;
006    
007    /**
008     * Thrown when a new chunk has finished being populated.
009     * <p>
010     * If your intent is to populate the chunk using this event, please see {@link
011     * BlockPopulator}
012     */
013    public class ChunkPopulateEvent extends ChunkEvent {
014        private static final HandlerList handlers = new HandlerList();
015    
016        public ChunkPopulateEvent(final Chunk chunk) {
017            super(chunk);
018        }
019    
020        @Override
021        public HandlerList getHandlers() {
022            return handlers;
023        }
024    
025        public static HandlerList getHandlerList() {
026            return handlers;
027        }
028    }
