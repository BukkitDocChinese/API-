001    package org.bukkit.inventory;
002    
003    import org.bukkit.block.DoubleChest;
004    
005    /**
006     * Interface to the inventory of a Double Chest.
007     */
008    public interface DoubleChestInventory extends Inventory {
009    
010        /**
011         * Get the left half of this double chest.
012         *
013         * @return The left side inventory
014         */
015        Inventory getLeftSide();
016    
017        /**
018         * Get the right side of this double chest.
019         *
020         * @return The right side inventory
021         */
022        Inventory getRightSide();
023    
024        DoubleChest getHolder();
025    }
